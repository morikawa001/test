const SHEET_NAME = 'ResearchEducation_Events'; const REQUIRED_HEADERS = [ '起案行No', '年度_YEAR_1', '開催日_DATE_1', 'テーマ(標題)_TITLE', '担当講師_SPEAKER', '共催_COHOST_1', '対象_SUBJECT_1', 'HP_URL_url_1', 'Zoom_ID_WEBINER_ID', 'Zoom_パスコード_PASSCODE', '起案1開催DATE_KIAN_1', '起案2資料_DATE_KIAN_2', '起案3受講証修了証_DATE_KIAN_3' ]; function onOpen() { SpreadsheetApp.getUi() .createMenu('研修会PM') .addItem('管理画面を開く', 'showApp') .addToUi(); } function showApp() { const html = HtmlService .createHtmlOutputFromFile('Index') .setWidth(1400) .setHeight(900); SpreadsheetApp.getUi().showModalDialog(html, '研修会準備プロジェクト管理'); } function getSheet_() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.getRange(1, 1, 1, REQUIRED_HEADERS.length).setValues([REQUIRED_HEADERS]); } const lastCol = Math.max(sheet.getLastColumn(), REQUIRED_HEADERS.length); const headerValues = sheet.getRange(1, 1, 1, lastCol).getValues()[0].map(v => String(v || '').trim()); if (!headerValues[0]) { sheet.getRange(1, 1, 1, REQUIRED_HEADERS.length).setValues([REQUIRED_HEADERS]); } return sheet; } function getHeaderMap_(headers) { const map = {}; headers.forEach((h, i) => { if (h) map[h] = i; }); return map; } function rowToObject_(row, headers) { const obj = {}; headers.forEach((h, i) => { if (h) obj[h] = row[i]; }); return obj; } function normalizeDateValue_(v) { if (Object.prototype.toString.call(v) === '[object Date]' && !isNaN(v)) { return Utilities.formatDate(v, Session.getScriptTimeZone(), 'yyyy-MM-dd'); } return v == null ? '' : v; } function getEvents() { const sheet = getSheet_(); const lastRow = sheet.getLastRow(); const lastCol = sheet.getLastColumn(); const headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0].map(v => String(v || '').trim()); const headerMap = getHeaderMap_(headers); const missingFields = REQUIRED_HEADERS.filter(h => !(h in headerMap)); if (lastRow < 2) { return { headers, items: [], missingFields }; } const values = sheet.getRange(2, 1, lastRow - 1, lastCol).getValues(); const items = values .filter(row => row.some(cell => String(cell || '').trim() !== '')) .map(row => { const obj = rowToObject_(row, headers); Object.keys(obj).forEach(key => { obj[key] = normalizeDateValue_(obj[key]); }); return obj; }); return { headers, items, missingFields }; } function saveEvent(item) { const sheet = getSheet_(); let headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0].map(v => String(v || '').trim()); REQUIRED_HEADERS.forEach(header => { if (!headers.includes(header)) { headers.push(header); } }); sheet.getRange(1, 1, 1, headers.length).setValues([headers]); const idHeader = '起案行No'; const idIndex = headers.indexOf(idHeader); if (idIndex < 0) { throw new Error('起案行No 列が見つかりません。'); } const lastRow = sheet.getLastRow(); const data = lastRow >= 2 ? sheet.getRange(2, 1, lastRow - 1, headers.length).getValues() : []; const idValue = String(item[idHeader] || '').trim(); if (!idValue) { throw new Error('起案行No は必須です。'); } const outputRow = headers.map(h => item[h] == null ? '' : item[h]); let targetRow = -1; for (let i = 0; i < data.length; i++) { if (String(data[i][idIndex] || '').trim() === idValue) { targetRow = i + 2; break; } } if (targetRow > -1) { sheet.getRange(targetRow, 1, 1, headers.length).setValues([outputRow]); } else { sheet.getRange(sheet.getLastRow() + 1, 1, 1, headers.length).setValues([outputRow]); } return getEvents(); }
